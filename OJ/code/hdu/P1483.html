int Judge(char *a, char *b, int len)
{
    int i= 0, count = 0, j = 0;
    while (i < len && j < len - 1)
    {
        if (a[i] == b[j])
        {
            i ++;
            j ++;
        }
        else
        {
            count ++;
            i ++;
        }
    }
    if (j == len - 1 && count == 0)
    {
        return 1;
    }
    if (count == 1)
    {
        return 1;
    }
    return 0;
}
