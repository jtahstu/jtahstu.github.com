__int64 ex_gcd(__int64 a,__int64 b,__int64 &x, __int64 &y)
{
    if(b == 0)
    {
        x=1;
        y=0;
        return a;
    }
    __int64 result = ex_gcd(b,a%b,x,y);
    __int64 t=x;
    x=y;
    y=t-a/b*y;
    return result;
}
