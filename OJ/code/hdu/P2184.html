long long  a[70];
int ans[5][70];
int x[5];
void solve(int A,int C,int n,long long m)
{
    if(m==a[n])
    {
        for(int i=n;i>=1;i--)
            ans[C][x[C]+i]=i;
        x[C]+=n;
        return ;
    }
    if(m>a[n]/2)

    {
        x[C]++;
        ans[C][x[C]]=n;
        solve( 6-A-C,C,n-1,m-a[n]/2-1);
    }
    else
    {
        x[A]++;
        ans[A][x[A]]=n;
        solve(A,6-A-C,n-1,m);
    }
    return ;
}
int main()
