long long a[20][20];

void change(int &x,int &y,int k)
{
    if(x == k)
    {
        if(y == k)
        {
            x++;
        }
        else y--;
    }
    else if(x == n+1-k)
    {
        if(y == n+1-k)
            x--;
        else y++;
    }
    else if(y == k)
        x++;
    else x--;
}
