const int MAXN=1000010;
const int HASH=1000007;
struct HASHMAP
{
    int head[HASH],next[MAXN],size;
    LL state[MAXN];
    void init()
    {
        size=0;
        memset(head,-1,sizeof(head));
    }
    bool check(LL val)
    {
        int h=(val%HASH+HASH)%HASH;
        for(int i=head[h];~i;i=next[i])
            if(state[i]==val)return true;
        return false;
    }
    int insert(LL val)
    {
        int h=(val%HASH+HASH)%HASH;
        for(int i=head[h];~i;i=next[i])
        {
            if(val==state[i])
                return 1;
        }
        state[size]=val;
        next[size]=head[h];
        head[h]=size++;
        return 0;
    }
