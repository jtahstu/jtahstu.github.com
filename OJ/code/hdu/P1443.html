int run(int a, int b)
{
    int cnt = 0;
    int sum = 2 * a;
    int index = b % (2 * a);
    if(index == 0) index = sum;

    while(index > a)
    {
        sum--;
        index = (index + b - 1) % sum;
        if(index == 0) index = sum;
        cnt ++;
    }
    return cnt;
}
