        {
            ans[i]=1;
            update(i,1);
            for(l=n;l>i;l--)

                if(str[l]==str[i])

                    a=sum(i);
                    b=sum(l-1);
                    t=(b-a)%mod;
                    while(t<0)    t+=mod;

                    update(l,1+t);
                }
            }
        }

        for(i=1;i<=n;i++)    hehe=(hehe+ans[i])%mod;
