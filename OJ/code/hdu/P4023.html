int a[16];

{
    int T;
    int iCase=0;
    int now;
    int Alice,Bob;
    scanf("%d",&T);
    while(T--)
    {
        iCase++;
        printf("Case #%d: ",iCase);
        for(int i=1;i<=15;i++)
          scanf("%d",&a[i]);
        Alice=Bob=0;
        now=1;
        if(a[15]%2==0)
        {
            Alice+=a[15]/2;
            Bob+=a[15]/2;
        }
        else
        {
            Alice+=a[15]/2+1;
            Bob+=a[15]/2;
            now=2;
        }
        int tempa=a[5]+a[6];
        int tempb=a[3]+a[4];
        if(tempa==tempb)
        {
            Alice+=tempa;
            Bob+=tempb;
        }
        else if(tempa=Alice)printf("Bob\n");
            else printf("Alice\n");
        }
        else
        {
            if(Alice>=Bob)printf("Alice\n");
            else printf("Bob\n");
        }
    }
    return 0;
}
