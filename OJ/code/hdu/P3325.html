{

  return x*y;
 else if(op==3 && y && x%y==0)
  return x/y;
 else if(op==4 && x && y%x==0)

 else

void DFS(int cnt)
{
 int i,j;
 int x,y,k;
 if(cnt==4)
 {
  for(i=0;i<4;i++)
   if(!visit[i] && !mp[arry[i]])
   {
    result[ind++]=arry[i];
    mp[arry[i]]=true;
   }
  return ;
 }
 for(i=0;i<4;i++)
 {
  if(visit[i])
   continue;

  {
   if(visit[j])
    continue;
   visit[j]=true;
   x=arry[i];
   y=arry[j];
   for(k=0;k<6;k++)
   {
    arry[i]=Work(x,y,k);
    DFS(cnt+1);
   }
   arry[i]=x;
   arry[j]=y;
   visit[j]=false;
  }
 }
}
