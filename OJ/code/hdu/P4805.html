double disLL(spt p1,spt p2,spt q1,spt q2) {
    spt p=q1-p1,u=p2-p1,v=q2-q1;
    double d=(u*u)*(v*v)-SQ(u*v);
    if(sign(d)==0)return disLP(q1,q2,p1);
    double s=((p*u)*(v*v)-(p*v)*(u*v))/d;
    return disLP(q1,q2,p1+u*s);
}
