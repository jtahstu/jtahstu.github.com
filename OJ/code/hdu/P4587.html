void Insert(int u,int v)
{
    edge[NE].v=v;
    edge[NE].next=head[u];
    head[u]=NE++;
}

void Tarjan(int u,int vv)
{
    low[u]=dfn[u]=++cnt;
    mark[u]=true;
    for(int i=head[u];i!=-1;i=edge[i].next){
        int v=edge[i].v;
        if(v==vv)continue;
        if(dfn[v]==0){
            Tarjan(v,vv);
            low[u]=min(low[u],low[v]);
            if(low[v]>=dfn[u]){
               blocks[u]++;
           }
        }else if(mark[v]){
            low[u]=min(low[u],dfn[v]);
        }
    }
}
