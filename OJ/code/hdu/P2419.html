multiset v[MAXN];
int pre[MAXN];

int root(int a)
{
    if(pre[a] != a)
        pre[a] = root(pre[a]);
    return pre[a];
}

int unionSet(int a, int b)
{
    multiset::iterator it;
    a = root(a), b = root(b);
    if(a == b)
        return 0;
    if(v[a].size() > v[b].size())
        swap(a, b);
    for    (it = v[a].begin(); it != v[a].end(); it++)
        v[b].insert(*it);
    v[a].clear();
    pre[a] = b;
    return 0;
}
