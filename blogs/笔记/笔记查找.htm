  1. int InsertBST(BSTNode *&p,KeyType k)     
  2. {  
  3.    if (p==NULL)                        //原树为空,新插入的记录为根结点  
  4.     {  
  5.         p=(BSTNode*)malloc(sizeof(BSTNode));  
  6.         p->key=k;  
  7.         p->lchild=p->rchild=NULL;  
  8.         return 1;  
  9.     }  
  10.     else if(k==p->key)                 //树中存在相同关键字的结点,返回0  
  11.         return 0;  
  12.     else if(k<p->key)  
  13.         returnInsertBST(p->lchild,k);  //插入到*p的左子树中  
  14.     else   
  15.         returnInsertBST(p->rchild,k);  //插入到*p的右子树中  
  16. }
